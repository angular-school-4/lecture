Destructuring

                
                const person = {name: 'Adam', age: 22};
                const {name, age} = person;
                const {name: imie, age: wiek} = person;

                console.log(name, age); // 'Adam 22'
                console.log(imie, wiek); // 'Adam 22'

                for (let index in array) {
                  console.log(index);
                });
                // 0 (**that will be string not number)
                // 1
                
            
                
                let array = [10, 20];
                for (var value of array) {
                  console.log(value);
                }
                // 10

Modules - imports

                      console.log(square(2));

                    console.log(utils.square(2));

TypeScript

  • open-source, developed by Microsoft,
  • upperset of JavaScript (JS script == working TS script),
  • transcompiles to JavaScript,
  • additional features:
    • static typing,
    • classes,
    • interfaces,
    • modules,
    • enums,
    • optional parameters.

narrowing

                  
                    function padLeft(padding: number | string, input: string) {
                      if (typeof padding === "number") {
                        return " ".repeat(padding) + input; // padding is a number
                      }
                      return padding + input; // padding can only be a string
                    }

Optional chaining

                
                let adam: Person | undefined

                console.log(adam.engine) //[ERR] Object is possibly 'undefined'

Nullish coalescing

                  
                    type Book = {title?: string, subtitle?: string, author?: string}

                    let customConfig: Book = {title: "TypeScript", subtitle: null}

                    console.log(customConfig.author ?? 'Default title')
                    // 'TypeScript'

                    console.log(customConfig.author ?? 'Unknown author')
                    // 'Unknown author'

                    console.log(customConfig.subtitle ?? 'Default subtitle')
                    // null
